#version 430

in vec2 worldPosition;
in float scale;
in float rotation;
in float imageIndex;

out float pass_scale;
out float pass_rotation;
out float pass_imageIndex;

void main(void) {
    gl_Position = vec4(worldPosition.xy, 0.0, 1.0);

    pass_scale = scale;
    pass_rotation = rotation;
    pass_imageIndex = imageIndex;
}
